  first=1
  for m in ~/resource_packs/*/manifest.json; do
    uuid=$(sed -n 's/.*"uuid"[[:space:]]*:[[:space:]]*"\([^"]*\)".*/\1/p' "$m" | head -n1)
    version=$(sed -n 's/.*"version"[[:space:]]*:[[:space:]]*\[\([^]]*\)\].*/[\1]/p' "$m" | head -n1)

    if [ -n "$uuid" ] && [ -n "$version" ]; then
      if [ $first -eq 0 ]; then echo ','; fi
      echo "  { \"pack_id\": \"$uuid\", \"version\": $version }"
      first=0
    fi
  done
  echo ']'
} > ~/worlds/world/world_resource_packs.json
